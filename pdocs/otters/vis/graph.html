259    def boundYAxis(self):
260        """
261        **Should be replaced by the folloifing:**
262        self.fig.update_yaxes(autorange="max", autorangeoptions_include=0)  
263
264        Bounds the range of each y axis to the traces currently present on the plot
265        This is way too complicated, but I need to move on
266        """
267        # Get all the y_maxes on the plot and their y axis
268        y_maxes = [{'yaxis':item['yaxis'], 'max':item['y'].max()} for item in self.fig.data]
269
270        # Create a dict of the y_maxes sorted by max. Then update another dict with the highest max for that axis, creating a "set" of the axes with the highest max
271        y_maxes_set = {}
272        y_maxes_dict = [{ymax['yaxis'] : ymax['max']} for ymax in sorted(y_maxes, key=lambda x: x['max'])]
273        for axis in y_maxes_dict:
274            y_maxes_set.update(axis)
275
276        # the primary axis is 'null' so that's how I'm checking for a secondary axis. 
277            # Notice this will only work with 2 axes, but who cares
278        for axis in y_maxes_set.items():
279            if axis[0]:
280                sec_y = True
281            else:
282                sec_y=False
283            self.fig.update_yaxes(range=[0,axis[1]], secondary_y=sec_y)
284        return

286    def scaleDTick(self):
287        dateDifference = (self.df.index[-1]-self.df.index[0]).days
288        if dateDifference < 1:
289            dTick=86400000.0/24
290        elif dateDifference < 2:
291            dTick=86400000.0*0.25
292        elif dateDifference < 7:
293            dTick=86400000.0*1
294        elif dateDifference < 14:
295            dTick=86400000.0*1
296        elif dateDifference < 24:
297            dTick='D1'
298        elif dateDifference < 32:
299            dTick=86400000.0*3
300        elif dateDifference < 168:
301            dTick=86400000.0*7
302        else:
303            dTick='M1'
304        return dTick
