140    def setDowntimeEvents(self, dropEventsBelowNRows=16, mergeWithinHours=12, mergeClose=True):
141        
142        if isinstance(self.df, pd.Series):
143            self.df = self.df.to_frame()
144        rowsBelowThreshold = self.df.applymap(lambda x: 1 if x and x > 0 and x < self.avgProd*(1-self.NPThreshold) else 0)
145        
146        eventsBelowThreshold = wrangler.gapAndIsland(rowsBelowThreshold)
147
148        downtimeEventDates = []
149        for event in eventsBelowThreshold:
150            if round(event.mean()) == 1 and len(event) >= dropEventsBelowNRows:
151                downtimeEventDates.append([event.index[0], event.index[-1]])
152
153        if mergeClose:
154            downtimeEventDates = wrangler.mergeCloseEvents(downtimeEventDates, mergeWithinHours=mergeWithinHours)
155
156        self.downtimeEvents = []
157        for event in downtimeEventDates:
158            self.downtimeEvents.append(DowntimeEvent(self, event))
159
160        return

465    def granulizeData(self, series, start, end):
466
467        for row in list(zip(series.index, series.values)):
468            now = start
469            while now <= end:
470                series.loc[row[0]+pd.Timedelta(hours=now)] = row[1]
471                now = now+1
