127    def setDowntimeEvents(self, dropEventsBelowNRows=16, mergeWithinHours=12, mergeClose=True):
128        
129        if isinstance(self.df, pd.Series):
130            self.df = self.df.to_frame()
131        rowsBelowThreshold = self.df.applymap(lambda x: 1 if x and x > 0 and x < self.avgProd*(1-self.NPThreshold) else 0)
132        
133        eventsBelowThreshold = wrangler.gapAndIsland(rowsBelowThreshold)
134
135        downtimeEventDates = []
136        for event in eventsBelowThreshold:
137            if round(event.mean()) == 1 and len(event) >= dropEventsBelowNRows:
138                downtimeEventDates.append([event.index[0], event.index[-1]])
139
140        if mergeClose:
141            downtimeEventDates = wrangler.mergeCloseEvents(downtimeEventDates, mergeWithinHours=mergeWithinHours)
142
143        self.downtimeEvents = []
144        for event in downtimeEventDates:
145            self.downtimeEvents.append(DowntimeEvent(self, event))
146
147        return

169        else:
170            self.targetPercentage = 1

172    def setAvgNonProd(self):
173        actualAvgRows = 16
174        dfCol = self.parent.df
175        colName = dfCol.squeeze().name
176
177        self.actualNPAvg = int(self.parent.getNRowAvg(dfCol.loc[(dfCol.index>=self.startTime) & (dfCol.index<=self.endTime), colName], rows=actualAvgRows, ascending=True))
178        self.targetNPAvg = int(self.parent.avgProd*(1-self.targetPercentage))

452    def granulizeData(self, series, start, end):
453
454        for row in list(zip(series.index, series.values)):
455            now = start
456            while now <= end:
457                series.loc[row[0]+pd.Timedelta(hours=now)] = row[1]
458                now = now+1
459            series.drop(row[0], inplace=True) # Clears the row you started with
460        return series

454
455        for row in list(zip(series.index, series.values)):
456            now = start
457            while now <= end:
458                series.loc[row[0]+pd.Timedelta(hours=now)] = row[1]

461        return series
